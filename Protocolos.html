    Protocolos (project v0.1.0) 

     defimpl String.Chars, for: Tuple do
   def to_string(tuple) do
     interior =
       tuple
       |> Tuple.to_list()
       |> Enum.map(&Kernel.to_string/1)
       |> Enum.join(", ")

     "{#{interior}}"
   end
 end

    Ejemplos 

     iex> to_string({3.14, "apple", :pie})
 "{3.14, apple, pie}"

 iex> to_string(4)
 "4"

 iex> to_string(4.5)
 "4.5"
