    Comprensiones (project v0.1.0)

    Ejemplos

    Ejemplo 1:

    iex(1)> list = [1,2,3,4,5,6]
iex(2)> comp1 = for x <- (list ++ [99]), do: (x - 1) + 100
[100, 101, 102, 103, 104, 105, 198]

    Ejemplo 2:

    iex(3)> comp2 = for {key,val} <- %{"Key1"=>"Val1","Key2"=>"Val2"}, do: key<>"->"<>val
["Key1->Val1", "Key2->Val2"]

    Ejemplo 3:

    iex(4)> comp3 = for n <- 1..3, times <- 1..n, do: {n,times}
[{1, 1}, {2, 1}, {2, 2}, {3, 1}, {3, 2}, {3, 3}]

    Ejemplo 4: Comprensión con Filtros

    iex(5)> comp4 = for x <- 1..1000, rem(x,7)==0,x<100, do: x
[7, 14, 21, 28, 35, 42, 49, 56, 63, 70, 77, 84, 91, 98]
