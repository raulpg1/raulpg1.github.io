    Ejemplos

    Ejemplo 1: If else

    iex> if nil do
...>   "Nunca se ejecutará este sentencia"
...> else
...>   "Acierto"
...> end

    Ejemplo 2: Unless

    iex> unless true do
...>   "Nunca se ejecutará esta sentencia"
...> end

    Ejemplo 3: Case statement

    iex> testvalue = 500
iex> case testvalue do
...>   200 -> true
...>   404 -> true
...>   _ -> false
...> end
false

    Ejemplo 4: Case con guardias

    iex> case {1, 2, 3} do
...>   {1, x, 3} when x > 0 ->
...>     "Correcto"
...>   _ ->
...>     "Falso"
...> end
"Correcto"

    Ejemplo 5: Cond

    iex> cond do
...>   2*3 == 8 -> "Esto es falso"
...>   String.lenght("hola") == 6 -> "Esto también"
...>   true -> "Ninguna condición es cierta"
...> end
"Ninguna condición es cierta"

    Ejemplo 6: With

    iex> with {:ok, nombre} <- Map.fetch(user, :nombre),
iex>      {:ok, apellido} <- Map.fetch(user, :apellido),
iex>      do: apellido <> ", " <> nombre
